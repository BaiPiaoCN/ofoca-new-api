import openai

# 1. 配置 Ofoca AI 的接口地址
openai.api_base = "https://coding.cnmnmsl.cloud/v1"

# 2. 填入在 Ofoca AI 后台生成的令牌 (Token)

    model="gpt-4o",
    messages=[
        {"role": "user", "content": "你好，Ofoca AI！"}
    ]
)

print(response.choices[0].message.content)
